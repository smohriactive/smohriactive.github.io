package org.apache.thermo.view.component
{
    import flash.events.Event;
    
    import mx.collections.ArrayCollection;
    
    import org.apache.thermo.data.vo.ViewVO;
    
    import spark.components.SkinnableContainer;
    import spark.components.TabBar;
    
    /**
     * @author masuland.com
     */
    public class ViewContainerComp extends SkinnableContainer
    {
        //----------------------
        // Skin Parts
        //----------------------
        
        [SkinPart("true")]
        public var tb_views:TabBar;
        
        //----------------------
        // Properties
        //----------------------
        
        private var _allViews:ArrayCollection;
        private var _noDesignViews:ArrayCollection = new ArrayCollection();
        private var _isViewsDirty:Boolean;
        
        private var _isDesignModeActive:Boolean;
        private var _isDesignModeActiveDirty:Boolean;
        
        //----------------------
        // Getter / Setter
        //----------------------
        
        public function set isDesignModeActive(value:Boolean):void
        {
            if (_isDesignModeActive != value)
            {
                _isDesignModeActive = value;
                _isDesignModeActiveDirty = true;
                invalidateProperties();
            }
        }
        
        public function set views(value:ArrayCollection):void
        {
            if (_allViews != value)
            {
                _allViews = value;
                
                _noDesignViews.removeAll();
                for each (var view:ViewVO in _allViews)
                {
                    if (!view.isDesignView)
                    {
                        _noDesignViews.addItem(view);
                    }
                }
                
                _isViewsDirty = true;
                invalidateProperties();
            }
        }
        
        //----------------------
        // Methods
        //----------------------

        override protected function getCurrentSkinState():String
        {
            return super.getCurrentSkinState();
        } 
        
        override protected function partAdded(partName:String, instance:Object):void
        {
            super.partAdded(partName, instance);
            
            if (instance == tb_views) 
            {
                tb_views.labelField = "title";
                tb_views.iconField = "icon";
            }
        }
        
        override protected function partRemoved(partName:String, instance:Object):void
        {
            super.partRemoved(partName, instance);
        }
        
        override protected function commitProperties():void
        {
            super.commitProperties();
            
            if (_isViewsDirty)
            {
                if (_allViews == null)
                {
                    this.visible = false;
                    this.includeInLayout = false;
                }
                else
                {
                    tb_views.dataProvider = _allViews;
                    this.visible = true;
                    this.includeInLayout = true;
                }
                _isViewsDirty = false;
            }
            
            if (_isDesignModeActiveDirty)
            {
                if (_isDesignModeActive)
                {
                    if (_allViews == null)
                    {
                        this.visible = false;
                        this.includeInLayout = false;
                    }
                    else
                    {
                        tb_views.dataProvider = _allViews;
                        this.visible = true;
                        this.includeInLayout = true;
                    }
                }
                else
                {
                    if (_noDesignViews.length > 0)
                    {
                        tb_views.dataProvider = _noDesignViews;
                        this.visible = true;
                        this.includeInLayout = true;
                    }
                    else
                    {
                        this.visible = false;
                        this.includeInLayout = false;
                    }
                }
                
                _isDesignModeActiveDirty = false;
            }
        }
    }
}
