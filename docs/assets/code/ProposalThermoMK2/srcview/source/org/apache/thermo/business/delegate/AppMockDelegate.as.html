            explorerView.title = "Explorer";
            explorerView.icon = "icons/explorer.png";
            
            var outlineView:ViewVO = new ViewVO();
            outlineView.title = "Outline";
            outlineView.icon = "icons/outline.png";
            
            var componentsView:ViewVO = new ViewVO();
            componentsView.title = "Components";
            componentsView.icon = "icons/components.png";

            timelineView.title = "Timeline";
            timelineView.isDesignView = true;
            
            var editorCompView:ViewVO = new ViewVO();
            editorCompView.title = "ExampleComponent.as";
            editorCompView.icon = "icons/as_file.png";
            
            var editorSkinView:ViewVO = new ViewVO();
            editorSkinView.title = "ExampleSkin.mxml";
            editorSkinView.icon = "icons/mxml_file.png";

            interactionsView.title = "Interactions";
            interactionsView.isDesignView = true;
            
            var alignView:ViewVO = new ViewVO();
            alignView.title = "Align";
            alignView.isDesignView = true;
            
            var propertiesView:ViewVO = new ViewVO();
            propertiesView.title = "Properties";
            propertiesView.icon = "icons/properties.png";

            var stylesView:ViewVO = new ViewVO();
            stylesView.title = "Styles";
            stylesView.isDesignView = true;

            designTimeDataView.title = "Design-Time Data";
            designTimeDataView.isDesignView = true;

            perspective = new PerspectiveVO();
            perspective.title = "Essentials";

            // Left-Top Views
            
            views = new ArrayCollection();
            views.addItem(explorerView);
            views.addItem(outlineView);
