            import org.apache.thermo.action.event.PerspectiveEvent;
            import org.apache.thermo.data.vo.PerspectiveVO;
            import org.apache.thermo.data.vo.SettingsVO;
            
            import spark.events.IndexChangeEvent;
            import spark.utils.TextFlowUtil;
            
            [MessageDispatcher]public var dispatcher:Function;
            
            [Bindable]
Subscribe('appModel', objectId='settings')]
            public var settings:SettingsVO;
            
            [Bindable]
Subscribe('appModel', objectId='currentPerspective')]
            public var currentPerspective:PerspectiveVO;

            [Bindable]
Subscribe('appModel', objectId='isDesignModeActive')]
            public var isDesignModeActive:Boolean;
            
            [Bindable]private var modes:ArrayCollection = new ArrayCollection([
                {label:"Source", icon:"icons/source-mode.png"}, {label:"Design", icon:"icons/design-mode.png"}
            ]);
            
            protected function bb_modes_changeHandler(event:IndexChangeEvent):void
            {
                if (event.newIndex == 0)
                {
                    dispatcher(new DesignModeEvent(DesignModeEvent.DEACTIVATE));
                }
                else
                {
                    dispatcher(new DesignModeEvent(DesignModeEvent.ACTIVATE));
                }
            }
            
            protected function tb_perspectives_changeHandler(event:IndexChangeEvent):void
            {
                dispatcher(new PerspectiveEvent(PerspectiveEvent.LOAD, PerspectiveVO( tb_perspectives.selectedItem )));
            }
        ]]>
