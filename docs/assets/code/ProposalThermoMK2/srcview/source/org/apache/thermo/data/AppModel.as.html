package org.apache.thermo.data
{
    import org.apache.thermo.data.vo.FileVO;
    import org.apache.thermo.data.vo.PerspectiveVO;
    import org.apache.thermo.data.vo.SettingsVO;
    import org.apache.thermo.data.vo.ViewVO;
    
    [Bindable]
    /**
     * @author masuland.com
     */
    public class AppModel
    {
        //----------------------
        // Properties
        //----------------------
        
        [Publish('appModel', objectId='settings')]
        public var settings:SettingsVO;
        
        [Publish('appModel', objectId='currentPerspective')]
        public var currentPerspective:PerspectiveVO;
        
        [Publish('appModel', objectId='currentView')]
        public var currentView:ViewVO;
        
        [Publish('appModel', objectId='currentFile')]
        public var currentFile:FileVO;
        
        [Publish('appModel', objectId='isDesignModeActive')]
        public var isDesignModeActive:Boolean;
        
        //----------------------
        // Getters/Setters
        //----------------------
    }
}
