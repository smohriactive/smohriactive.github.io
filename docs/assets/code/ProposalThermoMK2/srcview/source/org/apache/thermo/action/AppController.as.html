package org.apache.thermo.action
{
    import flash.events.IEventDispatcher;
    
    import mx.controls.Alert;
    import mx.core.FlexGlobals;
    import mx.events.ResourceEvent;
    import mx.events.StyleEvent;
    import mx.resources.ResourceManager;
    import mx.rpc.AsyncToken;
    import mx.rpc.Responder;
    import mx.rpc.events.FaultEvent;
    import mx.rpc.events.ResultEvent;
    import mx.styles.IStyleManager2;
    import mx.styles.StyleManager;
    
    import org.apache.thermo.action.event.AppEvent;
    import org.apache.thermo.action.event.DesignModeEvent;
    import org.apache.thermo.action.event.PerspectiveEvent;
    import org.apache.thermo.business.delegate.IAppDelegate;
    import org.apache.thermo.data.AppModel;
    import org.apache.thermo.data.vo.PerspectiveVO;
    import org.apache.thermo.data.vo.SettingsVO;
    
    /**
     * @author masuland.com
     */
    public class AppController
    {
        //----------------------
        // Properties
        //----------------------
        
        [MessageDispatcher]public var dispatcher:Function;
        
        [Inject]public var appModel:AppModel;
        
        [Inject]public var appDelegate:IAppDelegate;
        
        //----------------------
        // Methods
        //----------------------
        
        [Init]public function init():void
        {
            appModel.isDesignModeActive = true;
            
            // get settings
            dispatcher(new AppEvent(AppEvent.GET_SETTINGS));
        }
        
        [MessageHandler('AppEvent.GET_SETTINGS')]
        public function getSettings(event:AppEvent):void
        {
            var token:AsyncToken = appDelegate.getSettings();
            token.addResponder(new Responder(getSettings_resultHandler, getSettings_faultHandler));
        }
        
        [MessageHandler('DesignModeEvent.DEACTIVATE')]
        public function activateDesignMode(event:DesignModeEvent):void
        {
            appModel.isDesignModeActive = false;
        }
        
        [MessageHandler('DesignModeEvent.ACTIVATE')]
        public function deactivateDesignMode(event:DesignModeEvent):void
        {
            appModel.isDesignModeActive = true;
        }

        [MessageHandler('PerspectiveEvent.LOAD')]
        public function loadPerspective(event:PerspectiveEvent):void
        {
            appModel.currentPerspective = event.perspective;
        }
        
        //----------------------
        // Handler
        //----------------------
        
        protected function getSettings_resultHandler(event:ResultEvent):void
        {
            appModel.settings = SettingsVO( event.result );
            appModel.currentPerspective = PerspectiveVO( appModel.settings.perspectives.getItemAt(0) );
        }
        
        protected function getSettings_faultHandler(event:FaultEvent):void
        {
            Alert.show('getSettings_fault: ' + event.fault);
        }
    }
}
