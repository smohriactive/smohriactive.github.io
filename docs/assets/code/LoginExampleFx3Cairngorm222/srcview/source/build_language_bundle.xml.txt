<?xml version="1.0" encoding="utf-8"?>
<project 
	name="Build Language Bundle" 
	basedir="." 
	default="main"
	>
	
	<taskdef 
		resource="flexTasks.tasks"
		classpath="${basedir}/libs/flexTasks.jar"
		/>
 	
	<property 
		file="build.properties"
		/>
 
	<target 
		name="main"
		>
		
		<antcall 
			target="en_US"
			/>	
		
		<antcall 
			target="de_DE"
			/>
		
	</target>
	
	
	<target name="en_US">
		<mxmlc>
			<locale>en_US</locale>
			<source-path>../locale/{locale}</source-path>
			<include-resource-bundles>pdt</include-resource-bundles>
			<output>../bin-debug/resources_en_US.swf</output>
		</mxmlc>
		<copy todir="../bin-release/">
			<fileset dir="../bin-debug/">
				<include name="resources_en_US.swf"/>
			</fileset>
		</copy>
	</target>

	
	<target name="de_DE">
		<mxmlc>
			<locale>de_DE</locale>
			<source-path>../locale/{locale}</source-path>
			<compiler.library-path dir="${FLEX_HOME}/frameworks" append="false">
				<include name="libs" />
			</compiler.library-path>
 
			<include-resource-bundles>pdt</include-resource-bundles>
			<output>../bin-debug/resources_de_DE.swf</output>
		</mxmlc>
		<copy todir="../bin-release/">
			<fileset dir="../bin-debug/">
				<include name="resources_de_DE.swf"/>
			</fileset>
		</copy>
	</target>
	
</project>
